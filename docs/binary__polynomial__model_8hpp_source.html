207  BinaryPolynomialModel(const std::vector<IndexType> &variables,
208  const PolynomialKeyList<std::size_t> &poly_key_distance_list,
209  const PolynomialValueList<FloatType> &poly_value_list,
210  const Vartype vartype
211  ): vartype_(vartype) {
212  if (vartype_ == Vartype::NONE) {
213  throw std::runtime_error("Unknown vartype detected");
214  }
215 
216  if (poly_key_distance_list.size() != poly_value_list.size()) {
217  throw std::runtime_error("The sizes of key_list and value_list must match each other");
218  }
219 
220  variables_ = std::unordered_set<IndexType>(variables.begin(), variables.end());
221 
222  if (variables_.size() != variables.size()) {
223  throw std::runtime_error("Unknown error. It seems that the input variables contain the same variables");
224  }
225 
226  std::size_t num_interactions = poly_key_distance_list.size();
227  poly_key_list_.resize(num_interactions);
228  poly_value_list_.resize(num_interactions);
229 
230 #pragma omp parallel for
231  for (int64_t i = 0; i < (int64_t)num_interactions; ++i) {
232  std::vector<IndexType> temp;
233  for (const auto &it: poly_key_distance_list[i]) {
234  temp.push_back(variables[it]);
235  }
236  std::sort(temp.begin(), temp.end());
237  poly_key_list_[i] = temp;
238  poly_value_list_[i] = poly_value_list[i];
239  }
240 
241  for (std::size_t i = 0; i < num_interactions; ++i) {
242  poly_key_inv_[poly_key_list_[i]] = i;
243  for (const auto &it: poly_key_list_[i]) {
244  each_variable_num_[it]++;
245  }
246  }
247 
248  UpdateVariablesToIntegers();
249 
250  }

439  void RemoveInteraction(std::vector<IndexType> &key) {
440  FormatPolynomialKey(&key, vartype_);
441  if (poly_key_inv_.count(key) == 0) {
442  return;
443  }
444 
445  for (const auto &index: key) {
446  if (each_variable_num_[index] >= 2) {
447  each_variable_num_[index]--;
448  }
449  else if (each_variable_num_[index] == 1) {
450  each_variable_num_.erase(index);
451  variables_.erase(index);
452  relabel_flag_for_variables_to_integers_ = true;
453  }
454  }
455 
456  std::size_t inv = poly_key_inv_[key];
457 
458  std::swap(poly_key_inv_[key], poly_key_inv_[poly_key_list_.back()]);
459  poly_key_inv_.erase(key);
460 
461  std::swap(poly_key_list_[inv], poly_key_list_.back());
462  poly_key_list_.pop_back();
463 
464  std::swap(poly_value_list_[inv], poly_value_list_.back());
465  poly_value_list_.pop_back();
466 
467  }

519  void AddInteraction(std::vector<IndexType> &key, const FloatType &value, const Vartype vartype = Vartype::NONE) {
520  if (std::abs(value) <= 0.0) {
521  return;
522  }
523 
524  if (vartype_ == vartype || vartype == Vartype::NONE) {
525  FormatPolynomialKey(&key, vartype_);
526  SetKeyAndValue(key, value);
527  }
528  else {
529  const std::size_t original_key_size = key.size();
530  const std::size_t changed_key_list_size = IntegerPower(2, original_key_size);
531 
532  if (vartype_ == Vartype::SPIN && vartype == Vartype::BINARY) {
533  FormatPolynomialKey(&key, vartype);
534  for (std::size_t i = 0; i < changed_key_list_size; ++i) {
535  const auto changed_key = GenerateChangedKey(key, i);
536  int sign = ((original_key_size - changed_key.size())%2 == 0) ? 1.0 : -1.0;
537  SetKeyAndValue(changed_key, value*IntegerPower(2, changed_key.size())*sign);
538  }
539  }
540  else if (vartype_ == Vartype::BINARY && vartype == Vartype::SPIN) {
541  FormatPolynomialKey(&key, vartype);
542  FloatType changed_value = value*(1.0/changed_key_list_size);
543  for (std::size_t i = 0; i < changed_key_list_size; ++i) {
544  SetKeyAndValue(GenerateChangedKey(key, i), changed_value);
545  }
546  }
547  else {
548  throw std::runtime_error("Unknown vartype error");
549  }
550  }
551  }

609  FloatType Energy(const Sample<IndexType> &sample, bool omp_flag = true) const {
610  if (sample.size() != GetNumVariables()) {
611  throw std::runtime_error("The size of sample must be equal to num_variables");
612  }
613 
614  if (GetNumInteractions() == 0) {
615  return 0.0;
616  }
617 
618  std::size_t num_interactions = GetNumInteractions();
619  FloatType val = 0.0;
620 
621  if (omp_flag) {
622 #pragma omp parallel for reduction (+: val)
623  for (int64_t i = 0; i < (int64_t)num_interactions; ++i) {
624  int32_t spin_multiple = 1;
625  for (const auto &index: poly_key_list_[i]) {
626  spin_multiple *= sample.at(index);
627  if (spin_multiple == 0.0) {
628  break;
629  }
630  }
631  val += spin_multiple*poly_value_list_[i];
632  }
633  }
634  else {
635  for (std::size_t i = 0; i < num_interactions; ++i) {
636  int32_t spin_multiple = 1;
637  for (const auto &index: poly_key_list_[i]) {
638  spin_multiple *= sample.at(index);
639  if (spin_multiple == 0.0) {
640  break;
641  }
642  }
643  val += spin_multiple*poly_value_list_[i];
644  }
645  }
646  return val;
647  }

726  void Scale(const FloatType scalar,
727  const PolynomialKeyList<IndexType> &ignored_interactions = {},
728  const bool ignored_offset = false) {
729 
730  std::size_t num_interactions = GetNumInteractions();
731 
732  for (std::size_t i = 0; i < num_interactions; ++i) {
733  poly_value_list_[i] *= scalar;
734  }
735 
736  FloatType scalar_inv = 1.0/scalar;
737  for (const auto &key: ignored_interactions) {
738  if (poly_key_inv_.count(key) != 0) {
739  poly_value_list_[poly_key_inv_[key]] *= scalar_inv;
740  }
741  }
742 
743  if (ignored_offset == true && poly_key_inv_.count(std::vector<IndexType>{}) != 0 && std::count(ignored_interactions.begin(), ignored_interactions.end(), std::vector<IndexType>{}) == 0) {
744  poly_value_list_[poly_key_inv_[std::vector<IndexType>{}]] *= scalar_inv;
745  }
746  }

752  void normalize(const std::pair<FloatType, FloatType> &range = {1.0, 1.0},
753  const PolynomialKeyList<IndexType> &ignored_interactions = {},
754  const bool ignored_offset = false) {
755 
756  if (GetNumInteractions() == 0) {
757  return;
758  }
759 
760  FloatType max_poly_value = poly_value_list_[0];
761  FloatType min_poly_value = poly_value_list_[0];
762 
763  for (const auto &poly_value: poly_value_list_) {
764  if (max_poly_value < poly_value) {
765  max_poly_value = poly_value;
766  }
767  if (min_poly_value > poly_value) {
768  min_poly_value = poly_value;
769  }
770  }
771 
772  FloatType inv_scale = std::max(min_poly_value/range.first, max_poly_value/range.second);
773 
774  if (inv_scale != 0.0) {
775  Scale(1.0/inv_scale, ignored_interactions, ignored_offset);
776  }
777  }

835  Polynomial<IndexType, FloatType> ToHubo() const {
836  if (vartype_ == Vartype::BINARY) {
837  return GetPolynomial();
838  }
839  Polynomial<IndexType, FloatType> poly_map;
840  std::size_t num_interactions = GetNumInteractions();
841  for (std::size_t i = 0; i < num_interactions; ++i) {
842  const std::vector<IndexType> &original_key = poly_key_list_[i];
843  const FloatType original_value = poly_value_list_[i];
844  const std::size_t original_key_size = original_key.size();
845  const std::size_t changed_key_list_size = IntegerPower(2, original_key_size);
846 
847  for (std::size_t j = 0; j < changed_key_list_size; ++j) {
848  const auto changed_key = GenerateChangedKey(original_key, j);
849  int sign = ((original_key_size - changed_key.size())%2 == 0) ? 1.0 : -1.0;
850  FloatType changed_value = original_value*IntegerPower(2, changed_key.size())*sign;
851  poly_map[changed_key] += changed_value;
852  if (poly_map[changed_key] == 0.0) {
853  poly_map.erase(changed_key);
854  }
855  }
856  }
857  return poly_map;
858  }
